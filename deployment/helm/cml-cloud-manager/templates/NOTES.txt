1. Get the deployment name:
  kubectl get pods -l app.kubernetes.io/name={{ include "cml.name" . }} -n {{ .Release.Namespace }}

2. Port forward (if no ingress):
  kubectl port-forward svc/{{ include "cml.name" . }} 8000:{{ .Values.service.port }} -n {{ .Release.Namespace }}

3. Redis service:
  kubectl get svc {{ include "cml.name" . }}-redis -n {{ .Release.Namespace }}

4. MongoDB service:
  kubectl get svc {{ include "cml.name" . }}-mongodb -n {{ .Release.Namespace }}

5. Override sensitive values when installing:
  helm install my-cml charts/cml-cloud-manager \
    --set image.repository=ghcr.io/your-org/cml-cloud-manager,image.tag=1.0.0 \
    --set secretEnv.SESSION_SECRET_KEY=$(openssl rand -hex 32) \
    --set secretEnv.AWS_ACCESS_KEY_ID=YOUR_KEY \
    --set secretEnv.AWS_SECRET_ACCESS_KEY=YOUR_SECRET \
    --set mongodb.auth.password=securePass123

6. Update AWS credentials later:
  kubectl delete secret {{ include "cml.name" . }}-secret -n {{ .Release.Namespace }} && \
  kubectl create secret generic {{ include "cml.name" . }}-secret \
    --from-literal=AWS_ACCESS_KEY_ID=YOUR_KEY \
    --from-literal=AWS_SECRET_ACCESS_KEY=YOUR_SECRET \
    --from-literal=SESSION_SECRET_KEY=$(openssl rand -hex 32)

7. Enable ingress:
  --set ingress.enabled=true --set ingress.hosts[0].host=cml.example.com
